create database hoctap DEFAULT CHARACTER SET utf8 DEFAULT COLLATE utf8_general_ci;
use hoctap;
create table users(id int primary key not null auto_increment, first_name varchar(20)  not null,
 last_name varchar(40), mobile_phone varchar(15), bithday date, email varchar(100), facebook varchar(200),
 address varchar(200), school varchar(200), education_level int,rank int );

create table question_types(id int primary key not null auto_increment, acronym varchar(10), type varchar(100));
insert into question_types values(1,'hhkg','Hình học không gian');
insert into question_types values(2,'pt','Phương trình');
insert into question_types values(3,'hhp','Hình học phẳng');
create table questions(id int primary key not null auto_increment, question_name varchar(100),question_content varchar(1000), question_type int,
	author_id int, date_modify date,
	foreign key (question_type) references question_types(id), 
	foreign key (author_id) references users(id)
 );

 
// foreign key of author_id foreign key (author_id) references users(id)
create table exams(id int primary key not null auto_increment, description varchar(200), date_modify date, author_id int);

create table exams_questions(id int auto_increment, exam_id int not null, question_id int not null, primary key (exam_id, question_id),
	foreign key (exam_id) references exams(id),
	foreign key (question_id) references questions(id)
);
